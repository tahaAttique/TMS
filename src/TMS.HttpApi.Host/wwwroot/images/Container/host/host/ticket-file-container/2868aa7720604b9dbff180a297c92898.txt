CREATE TABLE Staff (
    StaffID INT PRIMARY KEY AUTO_INCREMENT,
    FullName VARCHAR(100) NOT NULL,
    PreferredName VARCHAR(50),
    DateOfBirth DATE NOT NULL,
    Gender ENUM('Male', 'Female', 'Non-Binary', 'Other') NOT NULL,
    ProfilePictureUrl VARCHAR(255),
    PhoneNumber VARCHAR(20) NOT NULL,
    AlternatePhoneNumber VARCHAR(20),
    EmailAddress VARCHAR(100) NOT NULL UNIQUE,
    ResidentialAddress TEXT NOT NULL,
    EmergencyContactName VARCHAR(100),
    EmergencyContactRelation VARCHAR(50),
    EmergencyContactPhone VARCHAR(20),
    Designation VARCHAR(100) NOT NULL,
    DepartmentID INT NOT NULL,
    EmployeeType ENUM('Full-Time', 'Part-Time', 'Contractual') NOT NULL,
    JoiningDate DATE NOT NULL,
    Experience INT DEFAULT 0,
    EmployeeCode VARCHAR(50) UNIQUE,
    SupervisorID INT,
    WorkHours VARCHAR(50),
    EmploymentStatus ENUM('Active', 'On Leave', 'Resigned', 'Retired') NOT NULL,
    Username VARCHAR(50) UNIQUE,
    PasswordHash VARCHAR(255),
    AccessLevel ENUM('Admin', 'Faculty', 'Staff') NOT NULL,
    AttendanceID INT UNIQUE,
    BiometricID VARCHAR(50),
    LeaveQuota INT DEFAULT 0,
    LeaveBalance INT DEFAULT 0,
    ShiftDetails VARCHAR(50),
    PerformanceRating FLOAT,
    Achievements TEXT,
    Appraisals TEXT,
    DisciplinaryActions TEXT,
    MedicalInformation TEXT,
    ExtraCurricularInvolvement TEXT,
    LanguagesSpoken TEXT,
    Notes TEXT,
    FOREIGN KEY (DepartmentID) REFERENCES Departments(DepartmentID),
    FOREIGN KEY (SupervisorID) REFERENCES Staff(StaffID)
);

CREATE TABLE FinancialDetails (

    FOREIGN KEY (StaffID) REFERENCES Staff(StaffID)

    CheckOutTime TIME,
    Status ENUM('Present', 'Absent', 'On Leave', 'Late') NOT NULL,

CREATE TABLE PerformanceAppraisals (
    AppraisalID INT PRIMARY KEY AUTO_INCREMENT,
    StaffID INT NOT NULL,
    AppraisalDate DATE NOT NULL,

Qualifications & Certifications Tables: Tracks academic and professional details.
Financial Details: Stores salary and payment-related information.

PerformanceAppraisals: Tracks performance evaluations.
